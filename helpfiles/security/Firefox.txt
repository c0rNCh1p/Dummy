  > Cursor Navigate : 'off' # dont navigate pages with cursor

  > New Homepage : 'duckduckgo.com' # set new homepage

  > Recent Activity : 'off' # dont show recent activity content on homepage

  > Search Bar : 'Add search bar in toolbar' # search bar next to address bar
